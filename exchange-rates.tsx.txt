import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { useExchangeRates } from "@/hooks/use-exchange-rates";
import { TrendingUp, TrendingDown } from "lucide-react";

export function ExchangeRates() {
  const { exchangeRates, isLoading, lastUpdated } = useExchangeRates();

  if (isLoading) {
    return (
      <Card data-testid="exchange-rates-loading">
        <CardHeader>
          <CardTitle className="flex items-center justify-between">
            Live Rates
            <div className="w-2 h-2 bg-muted rounded-full animate-pulse"></div>
          </CardTitle>
        </CardHeader>
        <CardContent>
          <div className="space-y-3">
            {[1, 2, 3, 4].map((i) => (
              <div key={i} className="flex justify-between items-center">
                <div className="h-4 bg-muted rounded w-20 animate-pulse"></div>
                <div className="h-4 bg-muted rounded w-16 animate-pulse"></div>
              </div>
            ))}
          </div>
        </CardContent>
      </Card>
    );
  }

  return (
    <Card data-testid="exchange-rates-card">
      <CardHeader>
        <CardTitle className="flex items-center justify-between">
          Live Rates
          <div className="w-2 h-2 bg-secondary rounded-full animate-pulse"></div>
        </CardTitle>
      </CardHeader>
      <CardContent>
        <div className="space-y-3">
          {exchangeRates.map((rate, index) => {
            const isPositive = rate.change24h >= 0;
            const Icon = isPositive ? TrendingUp : TrendingDown;
            
            return (
              <div key={rate.currency} className="flex items-center justify-between" data-testid={`exchange-rate-${index}`}>
                <div className="flex items-center space-x-2">
                  <span className="text-sm font-medium text-foreground" data-testid={`rate-currency-${index}`}>
                    {rate.currency}
                  </span>
                  <Badge 
                    variant={isPositive ? "secondary" : "destructive"}
                    className="text-xs"
                    data-testid={`rate-change-${index}`}
                  >
                    <Icon className="w-3 h-3 mr-1" />
                    {isPositive ? '+' : ''}{rate.change24h.toFixed(1)}%
                  </Badge>
                </div>
                <span className="font-semibold text-foreground" data-testid={`rate-value-${index}`}>
                  {rate.currency.includes('JPY') 
                    ? `¥${rate.rate.toLocaleString()}`
                    : rate.currency.includes('EUR')
                    ? `€${rate.rate.toFixed(2)}`
                    : rate.currency.includes('GBP')
                    ? `£${rate.rate.toFixed(2)}`
                    : `$${rate.rate.toFixed(2)}`
                  }
                </span>
              </div>
            );
          })}
        </div>
        {lastUpdated && (
          <div className="mt-4 pt-4 border-t border-border">
            <p className="text-xs text-muted-foreground text-center" data-testid="last-updated">
              Last updated: {new Date(lastUpdated).toLocaleTimeString()}
            </p>
          </div>
        )}
      </CardContent>
    </Card>
  );
}
